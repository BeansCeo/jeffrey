import axios from "axios";
const prompt = `Generate a quiz on the Periodic Table and Atomic Structure. You should include 5. The difficulty of the questions should be based on igce, stage 9.`;
  const apiKey = process.env.OPENAI_API_KEY;
    const client = axios.create({
      headers:{
        Authorization: "Bearer " + apiKey,
      },
    })
    const params = {
      prompt: `${prompt}`,
      model: "text-davinci-003",
      max_tokens: 10,
      temperature: 0, 
    };
    client
      .post("https://api.openai.com/v1/completions", params)
      .then((result)=>{
        let quiz = result.data.choices[0].text;
        console.log(quiz);
      })
    .catch((err) => {
        console.log(err);
      });
